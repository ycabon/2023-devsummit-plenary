NDVI 
const url =
  "https://tiledimageservices.arcgis.com/
  V6ZHFr6zdgNZuVG0/arcgis/rest/services/
  Glacier_national_park_2015/ImageServer";

const ndvi = new RasterFunction({
  functionName: "NDVI",
  functionArguments: {
    visibleBandID: 3,
    infraredBandID: 4,
    scientificOutput: false
  }
});

const colormap = new RasterFunction({
  functionName: "Colormap",
  functionArguments: {
    colormapName: NDVI3,
    raster: ndvi
  }
});

const layer = new ImageryTileLayer({
  url,
  rasterFunction: colormap
});
